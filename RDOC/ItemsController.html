def create
  @item = Item.new(params[:item])
  @search = Item.search(params[:search])

  if @item.save

    redirect_to admin_show_path(@item), notice: 'Item was successfully created.'

  else
    render action: "admin_new"
  end
end

def update
  @item = Item.find(params[:id])

  if @item.update_attributes(params[:item])
    redirect_to admin_show_path(@item), notice: 'Item was successfully updated.'
  else
    redirect_to admin_edit_path(@item)
  end
end
